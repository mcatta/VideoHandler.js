Video Handler Sample

Play | Pause 
var video = $('#video1').VideoHandler({
onReady: function() {
this.play();
},
onEnd: function() {
alert('Video finished');
},
onError: function() {
alert('Error to load video');
}
});

$('#play').on('click', function() {
video.play();
});
$('#stop').on('click', function() {
video.pause();
});
